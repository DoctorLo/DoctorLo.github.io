#include<cstdio>
#include<iostream>
using namespace std;
const int N = 30;
typedef long long ll;
int n,a[N],b[N],s[N],p[N][N];
ll dp[1<<23];
void solve(){
	for(int i = 1;i <= (1<<n+1);i++){
		/*先判断是否该状态合法,若合法则计算其值*/
		bool flag = true;
		for(int j = 1;j <= n;j++){
			if((1&(i>>j)) == 0) continue;
			for(int k = 1;k <= s[j];k++)
				if((1&(i>>p[j][k])) == 0){
					flag = false; break;
				} 
			if(!flag) break; int t = 0;
		}
		if(!flag) continue;
		for(int j = 1;j <= n;j++){
			if(!(i>>j&1)) continue;
			ll t = 0,s = i;
			while(s){
				if(s&1) t++; s >>= 1;
			}
			dp[i] = max(dp[i],dp[i^(1<<j)]+t*a[j]+b[j]);
		} 
	}
	printf("%lld\n",dp[(1<<n+1)-2]);
}
int main(){
	scanf("%d",&n);
	for(int i = 1;i <= n;i++){
		scanf("%d%d%d",a+i,b+i,s+i);
		for(int j = 1;j <= s[i];j++) scanf("%d",&p[i][j]);
	}
	solve();
	return 0;
}

J. Sum

原题链接
解题思路：
我想了一个O(9N)的算法，但是超时了，本机大概1.6-2.5s，很难受；是采用唯一分解定理判断每一个数可以分解为多少个“数对”相乘。然后改用线性筛，思想当然没变，但是利用线性筛的O(N)常数较小，本机0.7~1.1s，通过率大概1/4。
然后是朋友提供的解法思路，主要是统计“平方对”，不单独统计每个数对s的贡献，而是直接计算s，也是O(N)，但是更快，稳过。
